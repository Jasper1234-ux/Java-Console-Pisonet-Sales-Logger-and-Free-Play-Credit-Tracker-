<script>
function openLogin() { document.getElementById("login-modal").style.display="block"; }
function closeLogin() { document.getElementById("login-modal").style.display="none"; }

function login(){
    let username = document.querySelector("#login-modal input[type='text']").value;
    let password = document.querySelector("#login-modal input[type='password']").value;
    if(username==="admin" && password==="1234"){ alert("Welcome Admin!"); closeLogin(); }
    else if(username==="cashier" && password==="1234"){ alert("Welcome Cashier!"); closeLogin(); }
    else{ alert("Invalid Username or Password!"); }
}

/* STORAGE */
let sales = [];
let freePlays = [];

/* FUNCTIONS */
function logSale(){
    let amount = prompt("Enter sale amount:");
    if(amount) { sales.push(amount); alert("Sale recorded!"); }
}

function addFreePlay(){
    let minutes = prompt("Enter free-play minutes:");
    if(minutes) { freePlays.push(minutes); alert("Free-play recorded!"); }
}

function viewRecords(){
    let salesList = sales.length ? sales.join(", ") : "No sales yet";
    let freeList = freePlays.length ? freePlays.join(", ") : "No free-play yet";
    alert("SALES: " + salesList + "\nFREE-PLAY: " + freeList);
}

/* SHOW / HIDE PAGES */
function showStandalone(id){
    document.querySelectorAll('.standalone, #home').forEach(el=> el.style.display='none');
    document.getElementById(id).style.display = id==='home' ? '' : 'block';
}
</script>
